<script>
    let outfits = JSON.parse(localStorage.getItem('myWardrobeData')) || [];

    renderGrid();

    // --- 图片压缩逻辑 ---
    function compressImage(file, maxWidth = 1200, maxHeight = 1200, quality = 0.7) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.readAsDataURL(file);
            reader.onload = (event) => {
                const img = new Image();
                img.src = event.target.result;
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    let width = img.width;
                    let height = img.height;

                    if (width > height) {
                        if (width > maxWidth) {
                            height *= maxWidth / width;
                            width = maxWidth;
                        }
                    } else {
                        if (height > maxHeight) {
                            width *= maxHeight / height;
                            height = maxHeight;
                        }
                    }

                    canvas.width = width;
                    canvas.height = height;
                    const ctx = canvas.getContext('2d');
                    ctx.drawImage(img, 0, 0, width, height);

                    const dataUrl = canvas.toDataURL('image/jpeg', quality);
                    resolve(dataUrl);
                };
                img.onerror = error => reject(error);
            };
            reader.onerror = error => reject(error);
        });
    }

    async function addOutfit() {
        const imgInput = document.getElementById('imgInput');
        const top = document.getElementById('topInput').value.trim();
        const topLen = document.getElementById('topLenInput').value.trim();
        const bottom = document.getElementById('bottomInput').value.trim();
        const bottomLen = document.getElementById('bottomLenInput').value.trim();
        const btn = document.getElementById('addBtn');

        if (!imgInput.files || !imgInput.files[0]) {
            showToast("请选择一张图片！");
            return;
        }

        const file = imgInput.files[0];
        
        const originalBtnText = btn.innerHTML;
        btn.disabled = true;
        btn.innerHTML = '<span class="spinner"></span>压缩中...';

        try {
            const compressedDataUrl = await compressImage(file, 1200, 1200, 0.7);
            
            const newOutfit = {
                id: Date.now(),
                image: compressedDataUrl,
                top: top || "未填写",
                topLen: topLen || "", 
                bottom: bottom || "未填写",
                bottomLen: bottomLen || "", 
                date: new Date().toLocaleDateString(),
                originalSize: (file.size / 1024).toFixed(1) + 'KB',
                newSize: ((compressedDataUrl.length * 3 / 4) / 1024).toFixed(1) + 'KB'
            };

            outfits.unshift(newOutfit);
            saveAndRender();
            resetForm();
            showToast(`添加成功！已自动压缩 (${newOutfit.originalSize} → ${newOutfit.newSize})`);
        } catch (e) {
            console.error(e);
            showToast("图片处理失败，请重试");
        } finally {
            btn.disabled = false;
            btn.innerHTML = originalBtnText;
        }
    }

    function deleteOutfit(id) {
        if(confirm("确定要删除这条记录吗？")) {
            outfits = outfits.filter(item => item.id !== id);
            saveAndRender();
            showToast("已删除");
        }
    }

    function saveAndRender() {
        try {
            localStorage.setItem('myWardrobeData', JSON.stringify(outfits));
            renderGrid();
        } catch (e) {
            showToast("存储失败！空间已满，请删除一些旧记录。");
        }
    }

    function renderGrid() {
        const container = document.getElementById('gridContainer');
        const emptyState = document.getElementById('emptyState');
        
        container.innerHTML = '';

        if (outfits.length === 0) {
            emptyState.style.display = 'block';
            return;
        } else {
            emptyState.style.display = 'none';
        }

        outfits.forEach(item => {
            const topLenHtml = item.topLen ? `<span class="length-badge">${item.topLen}cm</span>` : '';
            const bottomLenHtml = item.bottomLen ? `<span class="length-badge">${item.bottomLen}cm</span>` : '';

            const card = document.createElement('div');
            card.className = 'outfit-card';
            card.innerHTML = `
                <button class="btn-delete" onclick="deleteOutfit(${item.id})" title="删除">×</button>
                <div class="card-image-wrapper" onclick="openLightbox('${item.image}')">
                    <img src="${item.image}" alt="Outfit" class="card-image">
                    <div class="zoom-overlay">
                        <div class="zoom-icon">🔍</div>
                        <div class="zoom-text">点击查看大图</div>
                    </div>
                </div>
                <div class="card-info">
                    <div class="size-tags">
                        <div class="tag-row">
                            <div class="tag-item">👕 ${item.top}</div>
                            ${topLenHtml}
                        </div>
                        <div class="tag-row">
                            <div class="tag-item">👖 ${item.bottom}</div>
                            ${bottomLenHtml}
                        </div>
                    </div>
                    <div style="font-size:0.75rem; color:#aaa; margin-top:10px; text-align:right;">
                        ${item.date} 
                        ${item.newSize ? `<span style="margin-left:5px; color:#ddd">(${item.newSize})</span>` : ''}
                    </div>
                </div>
            `;
            container.appendChild(card);
        });
    }

    function resetForm() {
        document.getElementById('imgInput').value = '';
        document.getElementById('topInput').value = '';
        document.getElementById('topLenInput').value = '';
        document.getElementById('bottomInput').value = '';
        document.getElementById('bottomLenInput').value = '';
    }

    function showToast(msg) {
        const toast = document.getElementById('toast');
        toast.textContent = msg;
        toast.classList.add('show');
        setTimeout(() => {
            toast.classList.remove('show');
        }, 3000);
    }

    // --- 增强版全屏查看器逻辑 (滚轮缩放 + 拖拽) ---
    const lightbox = document.getElementById('lightbox');
    const lightboxImg = document.getElementById('lightbox-img');
    const lightboxContainer = document.getElementById('lightboxContainer');

    let scale = 1;
    let pannedX = 0;
    let pannedY = 0;
    let isDragging = false;
    let startX, startY;
    const MIN_SCALE = 0.5;
    const MAX_SCALE = 5;

    function openLightbox(src) {
        // 重置状态
        scale = 1;
        pannedX = 0;
        pannedY = 0;
        updateTransform();
        
        lightboxImg.src = src;
        lightbox.classList.add('active');
        document.body.style.overflow = 'hidden';
    }

    function closeLightbox() {
        lightbox.classList.remove('active');
        setTimeout(() => {
            lightboxImg.src = '';
        }, 300);
        document.body.style.overflow = 'auto';
    }

    function updateTransform() {
        lightboxImg.style.transform = `translate(${pannedX}px, ${pannedY}px) scale(${scale})`;
    }

    // 1. 滚轮缩放
    lightboxContainer.addEventListener('wheel', (e) => {
        e.preventDefault();
        
        const rect = lightboxImg.getBoundingClientRect();
        // 计算鼠标相对于图片中心的偏移
        const x = e.clientX - rect.left - rect.width / 2;
        const y = e.clientY - rect.top - rect.height / 2;

        const delta = e.deltaY > 0 ? -0.1 : 0.1;
        const newScale = Math.min(Math.max(MIN_SCALE, scale + delta), MAX_SCALE);

        if (newScale !== scale) {
            // 简单的以中心缩放逻辑 (若要实现以鼠标为中心缩放逻辑会更复杂，这里采用中心缩放保证稳定性)
            // 如果需要更高级的以鼠标为中心，需要调整 pannedX/Y
            scale = newScale;
            
            // 如果缩放很小，自动归位
            if (scale <= 1) {
                pannedX = 0;
                pannedY = 0;
            }
            updateTransform();
        }
    }, { passive: false });

    // 2. 拖拽移动 (鼠标)
    lightboxContainer.addEventListener('mousedown', (e) => {
        if (scale <= 1) return; // 只有放大后才能拖拽
        isDragging = true;
        startX = e.clientX - pannedX;
        startY = e.clientY - pannedY;
        lightboxContainer.style.cursor = 'grabbing';
    });

    window.addEventListener('mousemove', (e) => {
        if (!isDragging) return;
        e.preventDefault();
        pannedX = e.clientX - startX;
        pannedY = e.clientY - startY;
        updateTransform();
    });

    window.addEventListener('mouseup', () => {
        isDragging = false;
        lightboxContainer.style.cursor = 'grab';
    });

    // 3. 双击重置
    lightboxContainer.addEventListener('dblclick', () => {
        scale = 1;
        pannedX = 0;
        pannedY = 0;
        updateTransform();
    });

    // 4. 触摸支持 (手机端简易拖拽)
    let touchStartX = 0, touchStartY = 0;
    
    lightboxContainer.addEventListener('touchstart', (e) => {
        if (e.touches.length === 1 && scale > 1) {
            isDragging = true;
            touchStartX = e.touches[0].clientX - pannedX;
            touchStartY = e.touches[0].clientY - pannedY;
        }
    }, { passive: false });

    lightboxContainer.addEventListener('touchmove', (e) => {
        if (isDragging && e.touches.length === 1) {
            e.preventDefault(); // 防止页面滚动
            pannedX = e.touches[0].clientX - touchStartX;
            pannedY = e.touches[0].clientY - touchStartY;
            updateTransform();
        }
    }, { passive: false });

    lightboxContainer.addEventListener('touchend', () => {
        isDragging = false;
    });

    // 键盘关闭
    document.addEventListener('keydown', function(event) {
        if (event.key === "Escape" && lightbox.classList.contains('active')) {
            closeLightbox();
        }
    });
    
    // 点击背景关闭 (排除点击图片本身)
    lightbox.addEventListener('click', (e) => {
        if (e.target === lightbox) {
            closeLightbox();
        }
    });

</script>
